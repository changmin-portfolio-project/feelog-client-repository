import React, { useEffect } from 'react';
import { useInView } from 'react-intersection-observer';
import styled from 'styled-components';

import { useRecoilState, useSetRecoilState } from 'recoil';
import { getPostSearch } from '../services/post/getPostSearch';
import {
  cursorIdAtomBySearchPost,
  isBookMarkedFavoriteTermAtom,
  searchPostHashMapAtom,
} from '../states/SearchPostAtom';

interface RepostInfiniteScrollProps {
  searchWord: string;
}

const SearchPostListInfiniteScrollBeta: React.FC<RepostInfiniteScrollProps> = ({
  searchWord,
}) => {
  const [cursorNum, setCursorNum] = useRecoilState(cursorIdAtomBySearchPost);

  const [ref, inView] = useInView();

  const [searchPostHashMap, setSearchPostHashMap] = useRecoilState(
    searchPostHashMapAtom,
  );

  const setIsBookMarkedFavoriteTerm = useSetRecoilState(
    isBookMarkedFavoriteTermAtom,
  );

  const callback = () => {
    getPostSearch(searchWord, cursorNum)
      .then((res) => {
        if (res.snsPostRspList.length > 0) {
          const newSearchPostHashMap = new Map(searchPostHashMap);

          res.snsPostRspList.forEach((snsPost) => {
            newSearchPostHashMap.set(snsPost.postId, snsPost);
          });

          setSearchPostHashMap(newSearchPostHashMap);
        }
        if (res.isFetchFavoriteState) {
          setIsBookMarkedFavoriteTerm(res.isBookMarkedFavoriteTerm);
        }

        setCursorNum(res.cursorId);
      })
      .catch((err) => {
        throw err;
      });
  };
  useEffect(() => {
    console.log('보인닷');
    console.log(searchWord);
    console.log(cursorNum);

    if (inView) {
      callback();
    }
  }, [inView, searchWord]);

  return (
    <ScrollBottomContainer ref={ref}>
      <div>검색 포스트 테스트</div>
    </ScrollBottomContainer>
  );
};

const ScrollBottomContainer = styled.div`
  margin: 0px auto;
`;

export default SearchPostListInfiniteScrollBeta;
